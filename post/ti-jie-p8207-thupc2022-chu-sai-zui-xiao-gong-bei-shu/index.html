	init();
	int cnt=0;

		int a=find(e[i].u),b=find(e[i].v);

	reg int x=0;reg char ch=getchar();

}

	if(vec[x].size()<vec[y].size())x^=y^=x^=y;

	return vec[y].clear(),x;
}
